- api/server/src/main/java/com/ke/bella/openapi/protocol/completion/ModelFallbackHandler.java: 这个文件实现了ModelFallbackHandler类，负责处理使用备用模型的完成请求。关键方法有handleCompletion，它尝试用主模型和备用模型处理请求，若全部失败则抛出FallbackException。它与ChatController、AdaptorManager、LimiterManager等多个组件交互。

- api/server/src/main/java/com/ke/bella/openapi/common/exception/FallbackException.java: 这个文件定义了FallbackException类，继承自RuntimeException，用于在所有备用模型尝试失败时抛出异常。它包含两个字段：attemptedModels和exceptions，分别记录尝试过的模型和对应的异常信息。

53     /**
54      * Handles the completion request with fallback models if specified
55      */
56     public CompletionResponse handleCompletion(CompletionRequest request, String endpoint, 
57                                               CompletionAdaptor<?> baseAdaptor, Object requestRiskData) {
58         request.parseModelFallbacks();
59         
60         // If no fallback models, just execute normally
61         if (CollectionUtils.isEmpty(request.getFallbackModels())) {
62             return executeCompletion(request, endpoint, baseAdaptor, requestRiskData);
63         }
64         
65         // Get timeout from header or use default
66         int timeoutSeconds = getTimeoutFromHeader();
67         long startTime = System.currentTimeMillis();
68         long endTime = startTime + TimeUnit.SECONDS.toMillis(timeoutSeconds);
69         
70         // Try the primary model first
71         Exception primaryException;
72         List<String> attemptedModels = new ArrayList<>();
73         attemptedModels.add(request.getModel());
74         
75         try {
76             return executeCompletion(request, endpoint, baseAdaptor, requestRiskData);
77         } catch (Exception e) {
78             log.warn("Primary model {} failed: {}", request.getModel(), e.getMessage());
79             
80             primaryException = e;
81             // Limit the number of fallback models to try
82             int modelsToTry = Math.min(request.getFallbackModels().size(), maxFallbackModels - 1);
83             
84             List<Exception> exceptions = new ArrayList<>();
85             // Try each fallback model
86             for (int i = 0; i < modelsToTry; i++) {
87                 // Check if we've exceeded the timeout
88                 if (System.currentTimeMillis() >= endTime) {
89                     throw new ChannelException.TimeoutException("Fallback models timeout after " + timeoutSeconds + " seconds");
90                 }
91                 
92                 String fallbackModel = request.getFallbackModels().get(i);
93                 attemptedModels.add(fallbackModel);
94                 
95                 try {
96                     // Create a copy of the request with the fallback model
97                     CompletionRequest fallbackRequest = JacksonUtils.convert(request, CompletionRequest.class);
98                     fallbackRequest.setModel(fallbackModel);
99                     fallbackRequest.setFallbackModels(null);
100                     
101                     // Execute with fallback model
102                     CompletionResponse response = executeCompletion(fallbackRequest, endpoint, baseAdaptor, requestRiskData);
103                     log.info("Successfully completed request with fallback model: {}", fallbackModel);
104                     return response;
105                 } catch (Exception fallbackEx) {
106                     log.warn("Fallback model {} failed: {}", fallbackModel, fallbackEx.getMessage());
107                     exceptions.add(fallbackEx);
108                 }
109             }
110             
111             // If all fallbacks failed, throw a FallbackException
112             throw new FallbackException("All models failed, including fallbacks", 
113                 attemptedModels, exceptions);
114         }
115     }
116     
117     private int getTimeoutFromHeader() {
118         String timeoutHeader = EndpointContext.getRequest().getHeader("X-BELLA-FALLBACK-TIMEOUT");
119         if (StringUtils.isNotEmpty(timeoutHeader)) {
120             try {
121                 int timeout = Integer.parseInt(timeoutHeader);
122                 return Math.max(1, Math.min(timeout, 180)); // Limit between 1-180 seconds
123             } catch (NumberFormatException e) {
124                 // Ignore parsing errors
125             }
126         }
127         return defaultTimeoutSeconds;
128     }
129

131                                               CompletionAdaptor<?> baseAdaptor, Object requestRiskData) {

101     private void updateBuffer(StreamCompletionResponse streamResponse) {
102         ResponseHelper.overwrite(responseBuffer, streamResponse);
103         if(CollectionUtils.isEmpty(streamResponse.getChoices()) || streamResponse.getChoices().get(0).getDelta() == null) {
104             return;
105         }
106         thinkStage = getThinkStage(streamResponse, thinkStage);
107         if(thinkStage == 1 || thinkStage == 3) {

119             choiceBuffer.get(choiceIndex).setFinish_reason(choice.getFinish_reason());

159     private void safetyCheck(boolean done) {
160         if(!dirtyChoice) {
161             return;
162         }
163         CompletionResponse.Choice choice = choiceBuffer.get(0);
164         if(!done) {
165             String content = thinkStage == 2 ? choice.reasoning() : choice.content();
166             String delta = content.substring(safetyCheckIndex);
