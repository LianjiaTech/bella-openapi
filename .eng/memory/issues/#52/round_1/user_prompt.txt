# 项目文件描述


- api/sdk/src/main/java/com/ke/bella/openapi/protocol/completion/CompletionRequest.java: 这个文件定义了CompletionRequest类，主要用于构建开放API的请求。它实现了UserRequest接口，并包含多个字段如model、user、messages等，用于配置API请求的参数和行为。

- api/server/src/main/java/com/ke/bella/openapi/protocol/completion/CompletionAdaptor.java: 定义CompletionAdaptor接口，用于处理补全请求，支持普通和流式补全。关联CompletionProperty类，用于异步或同步数据补全。

- api/server/src/main/java/com/ke/bella/openapi/endpoints/ChatController.java: ChatController类定义了用于处理聊天完成请求的REST接口，包含请求路由、安全检查、协议适配及响应逻辑。通过Spring框架实现REST控制器功能。

- api/server/src/main/java/com/ke/bella/openapi/protocol/ChannelRouter.java: 负责渠道路由选择的逻辑，包括筛选和选择可用的渠道，主要类是ChannelRouter，与ChannelService和ModelService交互。

- api/server/src/main/resources/application.yml: 该文件是Spring Boot应用程序的配置文件，设置了服务器、数据源、缓存、Swagger等配置，负责应用程序的初始参数。

72         String url = processData.getForwardUrl();
73         String channelInfo = channel.getChannelInfo();
74         CompletionAdaptor adaptor = adaptorManager.getProtocolAdaptor(endpoint, protocol, CompletionAdaptor.class);
75         CompletionProperty property = (CompletionProperty) JacksonUtils.deserialize(channelInfo, adaptor.getPropertyClass());
76         if(isMock) {
77             fillMockProperty(property);
78         }
79         adaptor = decorateAdaptor(adaptor, property, processData);
80 
81         EndpointContext.setEncodingType(property.getEncodingType());
82         if(request.isStream()) {
83             SseEmitter sse = SseHelper.createSse(1000L * 60 * 5, EndpointContext.getProcessData().getRequestId());

45     public ChannelDB route(String endpoint, String model, ApikeyInfo apikeyInfo, boolean isMock) {
46         if(StringUtils.isBlank(endpoint) && StringUtils.isBlank(model)) {
47             throw new BizParamCheckException("没有可用渠道");
48         }
49         List<ChannelDB> channels;
50         String entityCode;
51         if(model != null) {
52             String terminal = modelService.fetchTerminalModelName(model);
53             entityCode = terminal;
54             channels = channelService.listActives(EntityConstants.MODEL, terminal);
55         } else {
56             entityCode = endpoint;
57             channels = channelService.listActives(EntityConstants.ENDPOINT, endpoint);
58         }
59         if(CollectionUtils.isEmpty(channels)) {
60             if(isMock) {
61                 return mockChannel(null);
62             } else {
63                 throw new BizParamCheckException("没有可用渠道");
64             }
65         }
66         if(!isMock) {
67             channels = filter(channels, entityCode, apikeyInfo);
68         }
69         channels = pickMaxPriority(channels);
70         ChannelDB channel = random(channels);
71         return isMock ? mockChannel(channel) : channel;
72     }
73 
74     /**
75      * 1、筛选账户支持的数据流向（风控） 2、筛选可用的渠道
76      *
77      * @param channels
78      *
79      * @return
80      */
81     private List<ChannelDB> filter(List<ChannelDB> channels, String entityCode, ApikeyInfo apikeyInfo) {
82         Byte safetyLevel = apikeyInfo.getSafetyLevel();
83         String accountType = apikeyInfo.getOwnerType();
84         String accountCode = apikeyInfo.getOwnerCode();
85         List<ChannelDB> filtered = channels.stream()
86                 .filter(channel -> !EntityConstants.PRIVATE.equals(channel.getVisibility()) ||
87                         (accountType.equals(channel.getOwnerType()) && accountCode.equals(channel.getOwnerCode())))
88                 .filter(channel -> getSafetyLevelLimit(channel.getDataDestination()) <= safetyLevel)
89                 .collect(Collectors.toList());
90         if(CollectionUtils.isEmpty(filtered)) {
91             if(LOWEST_SAFETY_LEVEL.equals(safetyLevel)) {
92                 filtered = channels.stream().filter(this::isTestUsed)
93                         .collect(Collectors.toList());
94             }
95             if(CollectionUtils.isEmpty(filtered)) {
96                 throw new ChannelException.AuthorizationException("未经安全合规审核，没有使用权限");
97             }
98             if(freeAkOverload(EndpointContext.getProcessData().getAkCode(), entityCode)) {
99                 throw new ChannelException.RateLimitException("当前使用试用额度,每分钟最多请求" + freeRpm + "次, 且并行请求数不能高于" + freeConcurrent);
100             }
101         }
102         Set<String> unavailableSet = metricsManager.getAllUnavailableChannels(
103                 filtered.stream().map(ChannelDB::getChannelCode).collect(Collectors.toList()));
104         filtered = filtered.stream()
105                 .filter(channel -> channel.getDataDestination().equals(EntityConstants.PROTECTED) ||
106                         channel.getDataDestination().equals(EntityConstants.INNER) ||
107                         !unavailableSet.contains(channel.getChannelCode()))
108                 .collect(Collectors.toList());
109         if(CollectionUtils.isEmpty(filtered)) {
110             throw new ChannelException.RateLimitException("渠道当前负载过高，请稍后重试");
111         }
112         return filtered;
113     }
114 
115     private boolean isTestUsed(ChannelDB channel) {
116         return 1 == channel.getTrialEnabled() && !EntityConstants.PRIVATE.equals(channel.getVisibility());
117     }
118 
119     private boolean freeAkOverload(String akCode, String entityCode) {
120         return limiterManager.getRequestCountPerMinute(akCode, entityCode) >= freeRpm
121                 || limiterManager.getCurrentConcurrentCount(akCode, entityCode) >= freeConcurrent;
122     }
123 
124     private Byte getSafetyLevelLimit(String dataDestination) {
125         switch (dataDestination) {
126         case EntityConstants.PROTECTED:
127             return 10;
128         case EntityConstants.INNER:
129             return 20;
130         case EntityConstants.MAINLAND:
131             return 30;
132         case EntityConstants.OVERSEAS:
133             return 40;
134         }
135         return 40;
136     }
137 
138     private List<ChannelDB> pickMaxPriority(List<ChannelDB> channels) {
139         List<ChannelDB> highest = new ArrayList<>();
140         String curVisibility = EntityConstants.PUBLIC;
141         String curPriority = EntityConstants.LOW;
142         for (ChannelDB channel : channels) {
143             String priority = channel.getPriority();
144             String visibility = StringUtils.isNotBlank(channel.getVisibility()) ? channel.getVisibility() : EntityConstants.PUBLIC;
145             int compare = compare(priority, curPriority, visibility, curVisibility);
146             if(compare < 0) {
147                 continue;
148             }
149             if(compare > 0) {
150                 highest.clear();
151                 curPriority = priority;
152                 curVisibility = visibility;
153             }
154             highest.add(channel);
155         }
156         return highest;
157     }
158 
159     private int compare(String priority, String curPriority, String visibility, String curVisibility) {
160         if(visibility.equals(curVisibility)) {
161             if(priority.equals(curPriority)) {
162                 return 0;
163             }
164             if(priority.equals(EntityConstants.LOW)) {
165                 return -1;
166             }
167             if(priority.equals(EntityConstants.NORMAL)) {
168                 if(curPriority.equals(EntityConstants.HIGH)) {
169                     return -1;
170                 }
171             }
172             return 1;
173         } else {
174             return visibility.equals(EntityConstants.PRIVATE) ? 1 : -1;
175         }
176     }
177 
178     private ChannelDB random(List<ChannelDB> list) {
179         if(list.size() == 1) {
180             return list.get(0);
181         }
182         int rand = random.nextInt(list.size());
183         return list.get(rand);
184     }

194         channel.setSupplier("system");
195         channel.setUrl("");
196         return channel;
197     }
